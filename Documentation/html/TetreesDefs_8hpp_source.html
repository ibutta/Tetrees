494  void transpose()
495  {
496  unsigned r, c;
497  r = c = 0;
498  Type auxMatrixArray[rHeight * cWidth];
499  // Copies the main matrixArray to an auxiliary one.
500  for(unsigned auxR = 0; auxR < rHeight; auxR++){
501  for(unsigned auxC = 0; auxC < cWidth; auxC++){
502  auxMatrixArray[auxR * cWidth + auxC] = matrixArray[auxR * cWidth + auxC];
503  }
504  }

506  for(unsigned auxR = 0; auxR < rHeight; auxR++){
507  for(unsigned auxC = 0; auxC < cWidth; auxC++){
