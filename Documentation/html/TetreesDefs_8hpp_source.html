489  void transpose()
490  {
491  unsigned r, c;
492  r = c = 0;
493  Type auxMatrixArray[rHeight * cWidth];
494  // Copies the main matrixArray to an auxiliary one.
495  for(unsigned auxR = 0; auxR < rHeight; auxR++){
496  for(unsigned auxC = 0; auxC < cWidth; auxC++){
497  auxMatrixArray[auxR * cWidth + auxC] = matrixArray[auxR * cWidth + auxC];
498  }
